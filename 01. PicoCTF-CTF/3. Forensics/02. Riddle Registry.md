````
import requests
from urllib.parse import urljoin, urlparse
import re
import os

url = "https://challenge-files.picoctf.net/c_amiable_citadel/a8aa03694837741eed59c479749fc7f5bfd14fa66f4295b83776f16b2003a67d/confidential.pdf"

# Cookie guesses (try these)
cookies_list = [
    {"CookieMonster": "nom_nom"},
    {"CookieMonster": "give_me_cookies"},
    {"access": "granted"},
    {}  # no cookies as fallback
]

def get_full_source(url, cookies):
    """Fetch HTML and extract linked CSS/JS"""
    try:
        response = requests.get(url, cookies=cookies, timeout=10)
        if response.status_code != 200:
            print(f"Status {response.status_code} with cookies {cookies}")
            return None, None
        
        html = response.text
        base_dir = "web_source"
        os.makedirs(base_dir, exist_ok=True)
        
        # Save HTML
        with open(f"{base_dir}/index.html", "w", encoding="utf-8") as f:
            f.write(html)
        print(f"[+] Saved HTML to {base_dir}/index.html")
        
        # Find all CSS files
        css_links = re.findall(r'<link[^>]*href=["\']([^"\']+\.css[^"\']*)["\']', html)
        css_imports = re.findall(r'@import\s+url\(["\']([^"\']+\.css[^"\']*)["\']\)', html)
        all_css = css_links + css_imports
        
        # Find all JS files
        js_srcs = re.findall(r'<script[^>]*src=["\']([^"\']+\.js[^"\']*)["\']', html)
        
        # Also find inline JS
        inline_js = re.findall(r'<script[^>]*>([\s\S]*?)</script>', html)
        
        # Download CSS files
        for css_url in all_css:
            full_css_url = urljoin(url, css_url)
            try:
                css_response = requests.get(full_css_url, cookies=cookies, timeout=10)
                if css_response.status_code == 200:
                    filename = os.path.basename(urlparse(full_css_url).path) or "style.css"
                    filepath = f"{base_dir}/{filename}"
                    with open(filepath, "w", encoding="utf-8") as f:
                        f.write(css_response.text)
                    print(f"[+] Saved CSS: {filename}")
                else:
                    print(f"[-] Failed CSS: {css_url} (status {css_response.status_code})")
            except Exception as e:
                print(f"[-] Error downloading CSS {css_url}: {e}")
        
        # Download JS files
        for js_url in js_srcs:
            full_js_url = urljoin(url, js_url)
            try:
                js_response = requests.get(full_js_url, cookies=cookies, timeout=10)
                if js_response.status_code == 200:
                    filename = os.path.basename(urlparse(full_js_url).path) or "script.js"
                    filepath = f"{base_dir}/{filename}"
                    with open(filepath, "w", encoding="utf-8") as f:
                        f.write(js_response.text)
                    print(f"[+] Saved JS: {filename}")
                else:
                    print(f"[-] Failed JS: {js_url} (status {js_response.status_code})")
            except Exception as e:
                print(f"[-] Error downloading JS {js_url}: {e}")
        
        # Save inline JS
        if inline_js:
            with open(f"{base_dir}/inline_scripts.js", "w", encoding="utf-8") as f:
                for i, script in enumerate(inline_js):
                    f.write(f"// Inline Script {i+1}\n{script}\n\n")
            print(f"[+] Saved {len(inline_js)} inline script(s) to {base_dir}/inline_scripts.js")
        
        # Also check for inline CSS in <style> tags
        inline_css = re.findall(r'<style[^>]*>([\s\S]*?)</style>', html)
        if inline_css:
            with open(f"{base_dir}/inline_styles.css", "w", encoding="utf-8") as f:
                for i, css in enumerate(inline_css):
                    f.write(f"/* Inline Style {i+1} */\n{css}\n\n")
            print(f"[+] Saved {len(inline_css)} inline style block(s) to {base_dir}/inline_styles.css")
        
        return html, base_dir
        
    except Exception as e:
        print(f"[-] Error fetching {url}: {e}")
        return None, None

# Try different cookies
for cookies in cookies_list:
    print(f"\n[*] Trying cookies: {cookies}")
    print("=" * 60)
    html, base_dir = get_full_source(url, cookies)
    if html:
        print(f"\n[✓] Success! All files saved in '{base_dir}/' directory")
        
        # Quick check for flag in source
        if "picoCTF{" in html:
            flags = re.findall(r'picoCTF\{[^}]+\}', html)
            print(f"\n[!!!] FLAG FOUND in HTML: {flags}")
        break
    else:
        print(f"[✗] Failed with cookies: {cookies}")
        continue

print("\n" + "=" * 60)
print("To view all saved files, check the 'web_source' folder")
print("Open index.html in browser to see the full page")
````
